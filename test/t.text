/*
This is an example on how to use the 2.2" and 2.4" TFT 240x320 SPI ILI9341 display using the Adafruit library.

ILI9341 TFT SPI display pins for Arduino Uno/Nano:
This is a 3.3V device so you need to put a level shifter on all 5 data pins.
An easy way to do this is to use a 2.2K resistor between the data pin on the display to the pin on arduino, and
another resistor on the side of the display to a 3.3K resistor to ground. This creates a voltage divider to
convert the 5V output of arduino into a 3V logic. Do this for all 5 data pins.
 * LED =   10Î© (or more) resistor to 5V
 * SCK =   13
 * SDI =   11
 * D/C =   9
 * RESET = 8 or RST or 3.3V
 * CS =    10
 * GND =   GND
 * VCC =   5V or 3.3V (the display has it's own 3.3V regulator)

Hardware SPI Pins:
 * Arduino Uno   SCK=13, SDA=11
 * Arduino Nano  SCK=13, SDA=11
 * Arduino Due   SCK=76, SDA=75
 * Arduino Mega  SCK=52, SDA=51

SPI pin names can be confusing. These are the alternative names for the SPI pins:
MOSI = DIN = R/W = SDO = DI = SI = MTSR = SDA = D1 = SDI
CS = CE = RS = SS
DC = A0 = DO = DOUT = SO = MRST
RESET = RST
SCLK = CLK = E = SCK = SCL = D0


Libraries needed:
https://github.com/adafruit/Adafruit_ILI9341
https://github.com/adafruit/Adafruit-GFX-Library


Reference page for GFX Library: https://cdn-learn.adafruit.com/downloads/pdf/adafruit-gfx-graphics-library.pdf


Color is expressed in 16 bit with Hexadecimal value.
To select a particular color, go here and copy the "Hexadecimal 16 bit color depth value":
https://ee-programming-notepad.blogspot.com/2016/10/16-bit-color-generator-picker.html

Common colors:
 * BLACK    0x0000
 * BLUE     0x001F
 * RED      0xF800
 * GREEN    0x07E0
 * CYAN     0x07FF
 * MAGENTA  0xF81F
 * YELLOW   0xFFE0
 * WHITE    0xFFFF
#define TFT_YELLOW 0xFFE0
#define GREEN 0xFFE0
#define TFT_YELLOW 0xFFE0
#define TFT_YELLOW 0xFFE0
#define TFT_YELLOW 0xFFE0
#define TFT_YELLOW 0xFFE0

List of custom fonts: https://learn.adafruit.com/adafruit-gfx-graphics-library/using-fonts

Note about custom font:
 * Text background color is not supported for custom fonts. For these reason you would need to draw a filled 
   rectangle before drawing the text. But this would cause the text to flicker, so I don't recommend using custom fonts
   for components that refresh continuously.
 * Using custom fonts slows down the arduino loop, so the refresh rate is lesser than using the standard font.


Sketch made by: InterlinkKnight
Last modification: 01/29/2018
*/

#define TFT_DC 9

const byte ROWS = 4;

void clearScreen(uint16_t color) {
    tft.fillScreen(color);
}
void displayText(const char* text, int x, int y, uint8_t fontSize, uint16_t color) {
    tft.setTextColor(color);
    tft.setTextSize(fontSize);
    tft.setCursor(x, y);
    tft.print(text);
};
// Function to estimate text width
int16_t calculateTextWidth(const char* text, int16_t charWidth, int16_t fontSize) {
    return strlen(text) * 23;
}
void displayCenteredText(const char* text, int y, uint8_t fontSize, uint16_t color) {
    // Set the text size and color
    tft.setTextSize(fontSize);
    tft.setTextColor(color);
    
    // Calculate the width of the text in pixels
    int16_t textWidth = calculateTextWidth(text,2,fontSize);

    // Calculate the starting X coordinate to center the text
    int startX = (tft.width() - textWidth) / 2;

    // Display the text at the calculated position
    tft.setCursor(startX, y);
    tft.print(text);
}

bool confirmInput() {
    // Clear the screen and set up the prompt
    tft.setFont(&FreeSansBold9pt7b);
    tft.drawBitmap(0, 0, epd_bitmap_Background240x320, 320, 240, TFT_WHITE);
    displayWiFiSignal(50,280, 5);
    displayText("Save the entry?", 15,80, 2, TFT_YELLOW); // Display the main prompt
    tft.setFont(&FreeSansOblique9pt7b);
    // Display the options for confirmation
    displayText("No '*'", 15, 200, 2, TFT_RED); // Display 'Yes' option
    displayText("Yes '#'", 180, 200, 2, TFT_GREEN); // Display 'No' option
    while (true) {
        // Wait for key press input
        char key = keyPad.getKey();

        // Return true if the user confirms with '*'
        if (key == '#') {
            return true;
        }

        // Return false if the user declines with '#'
        if (key == '*') {
            return false;
        }

        delay(50); // Short delay to avoid rapid polling
    }
}
void eraseCharacter(int x, int y, uint16_t bgColor) {
    // Define the width and height of a character based on font size
    int charWidth = 20;   // Approximate character width, may vary with font library
    int charHeight = 60;  // Approximate character height

    // Draw a filled rectangle over the character area to "erase" it
    tft.fillRect(x, y, charWidth, charHeight, bgColor);
}
String ShowPrompt() {
    String promptInput = "";          // Stores the user's input
    uint16_t cursorPosX = 50;         // Initial cursor position for text
    uint16_t cursorPosY = 150;        // Initial cursor position for text
    uint8_t fontSize = 2;             // Font size for displayed text
    int width = 300; 
    int height = 60;
    tft.setFont(&FreeSansBold9pt7b);
    tft.drawBitmap(0, 0, epd_bitmap_Background240x320, 320, 240, TFT_WHITE);
    displayWiFiSignal(100, 280, 5);
    displayText("Karnataka", 10, 25, 1, TFT_WHITE);
    displayText("Enter Number :", 100,30+height, 1, 0xfbf1); // Display the main prompt
    int x = (tft.width() - width) / 2;
    int y = 20 + (tft.height() - height) / 2;
    tft.fillRect(x, y, width, height, 0xdb22);
    //drawCenteredBox(width, height, TFT_BLUE); // Draw a centered box for user input
    tft.setFont(&FreeSansBold9pt7b);

    while (true) {
        char character = keyPad.getKey(); // Get the character from the key press

        // Check if a valid character is pressed and process it
        if (character != NO_KEY) {
            if (promptInput.length() < 10) { // Limit input to 10 characters
                // Only append if it's a valid numeric character
                if (character >= '0' && character <= '9') { 
                    promptInput += character;                 // Append the character to the input string
                    displayText(&character, cursorPosX, cursorPosY, fontSize, TFT_WHITE); // Display character on the screen
                    cursorPosX +=23; // Move cursor for the next character
                    tft.setCursor(cursorPosX, cursorPosY);             // Update cursor position
                }
            }
        }

        // Check for the '*' key to delete the last character
        if (character == '*') {
          Serial.println("*");
           // if (!promptInput.isEmpty()) 
            {
                cursorPosX -=23; // Adjust cursor position back
                eraseCharacter(cursorPosX, y,TFT_BLUE); // Erase the last character
                promptInput.remove(promptInput.length() - 1); // Remove the last character from input
                tft.setCursor(cursorPosX, cursorPosY);             // Update cursor position
            }
            delay(20); // Short delay to prevent repeated deletions
        }

        // Check for the '#' key to confirm the input
        if (character == '#') {
          Serial.println("#");
            if (confirmInput()) {
                return promptInput; // Return the collected input if confirmed
            } else {
                return ""; // Return empty string if not confirmed
            }
        }

        delay(50); // Short delay to avoid rapid polling of key presses
    }
}

        delay(200); // Show message in red
        clearScreen(TFT_WHITE);
        displayCenteredText(message, tft.height() / 2, 2, TFT_RED);
        delay(200); // Clear screen to white for blink effect
        
    }
}

void drawArc(int x, int y, int r, uint16_t color) {
    for (int angle = 0; angle <= 180; angle++) { // Draw half arcs
        float rad = angle * 3.14159 / 180.0;
        int x0 = x + r * cos(rad);
        int y0 = y - r * sin(rad);
        tft.drawPixel(x0, y0, color); // Draw the arc pixel
    }
}
