#include <Keypad.h>

// Define row and column pins for the keypad
#define KEYPAD_ROW_1_PIN 12
#define KEYPAD_ROW_2_PIN 13
#define KEYPAD_ROW_3_PIN 14
#define KEYPAD_ROW_4_PIN 27

#define KEYPAD_COL_1_PIN 32
#define KEYPAD_COL_2_PIN 33
#define KEYPAD_COL_3_PIN 25

// Define the number of rows and columns
const byte ROW_NUM = 4;
const byte COLUMN_NUM = 3;

// Define the keys on the keypad
char keys[ROW_NUM][COLUMN_NUM] = {
    {'1', '2', '3'},
    {'4', '5', '6'},
    {'7', '8', '9'},
    {'*', '0', '#'}
};

// Define the row and column pin arrays
byte RowPins[ROW_NUM] = {KEYPAD_ROW_1_PIN, KEYPAD_ROW_2_PIN, KEYPAD_ROW_3_PIN, KEYPAD_ROW_4_PIN};
byte ColPins[COLUMN_NUM] = {KEYPAD_COL_1_PIN, KEYPAD_COL_2_PIN, KEYPAD_COL_3_PIN};

// Create the Keypad object
Keypad keypad = Keypad(makeKeymap(keys), RowPins, ColPins, ROW_NUM, COLUMN_NUM);

void setup() {
  Serial.begin(115200);
  Serial.println("Keypad test ready");
}

void loop() {
  char key = keypad.getKey();  // Check if a key is pressed

  if (key) {  // If a key is pressed, display it
    Serial.print("Key Pressed: ");
    Serial.println(key);
  }
}
