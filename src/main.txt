#include <SPI.h>
#include <MFRC522.h>
#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <Keypad.h>

// RFID Pins
#define RFID_RST_PIN 15
#define RFID_SDA_PIN 5
#define RFID_MISO_PIN 19
#define RFID_MOSI_PIN 23
#define RFID_SCK_PIN 18

// Keypad Pins
#define KEYPAD_ROW_1_PIN 4
#define KEYPAD_ROW_2_PIN 26
#define KEYPAD_ROW_3_PIN 27
#define KEYPAD_ROW_4_PIN 14
#define KEYPAD_COL_1_PIN 32
#define KEYPAD_COL_2_PIN 33
#define KEYPAD_COL_3_PIN 25

// LCD I2C Pins
#define SDA_PIN 21
#define SCL_PIN 22

// Keypad Configuration
const byte ROWS = 4;
const byte COLS = 3;
char keys[ROWS][COLS] = {
  {'1', '2', '3'},
  {'4', '5', '6'},
  {'7', '8', '9'},
  {'*', '0', '#'}
};
byte rowPins[ROWS] = {KEYPAD_ROW_1_PIN, KEYPAD_ROW_2_PIN, KEYPAD_ROW_3_PIN, KEYPAD_ROW_4_PIN};
byte colPins[COLS] = {KEYPAD_COL_1_PIN, KEYPAD_COL_2_PIN, KEYPAD_COL_3_PIN};

// Initialize objects
MFRC522 rfid(RFID_SDA_PIN, RFID_RST_PIN);
Keypad keypad = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS);
LiquidCrystal_I2C lcd(0x27, 16, 2); // Adjust address (0x27) based on your I2C LCD module

void setup() {
  // Serial and SPI
  Serial.begin(115200);
  SPI.begin(RFID_SCK_PIN, RFID_MISO_PIN, RFID_MOSI_PIN);

  // RFID
  rfid.PCD_Init();

  // LCD
  Wire.begin(SDA_PIN, SCL_PIN);
  lcd.init();
  lcd.backlight();
  lcd.setCursor(0, 0);
  lcd.print("Initializing...");
  delay(1000);

  // Print ready message
  lcd.clear();
  lcd.print("Ready for input");
  Serial.println("Place your RFID card near the reader or press a keypad key...");
}

void loop() {
  // RFID Scanning
  if (rfid.PICC_IsNewCardPresent() && rfid.PICC_ReadCardSerial()) {
    lcd.clear();
    lcd.print("Card UID:");
    Serial.print("Card UID: ");
    lcd.setCursor(0, 1);
    for (byte i = 0; i < rfid.uid.size; i++) {
      Serial.print(rfid.uid.uidByte[i] < 0x10 ? " 0" : " ");
      Serial.print(rfid.uid.uidByte[i], HEX);
      lcd.print(rfid.uid.uidByte[i] < 0x10 ? " 0" : " ");
      lcd.print(rfid.uid.uidByte[i], HEX);
    }
    Serial.println();
    rfid.PICC_HaltA();
    delay(2000); // Pause for 2 seconds
    lcd.clear();
    lcd.print("Ready for input");
  }

  // Keypad Input
  char key = keypad.getKey();
  if (key) {
    lcd.clear();
    lcd.print("Key Pressed:");
    lcd.setCursor(0, 1);
    lcd.print(key);
    Serial.print("Key Pressed: ");
    Serial.println(key);
    delay(1000); // Pause for 1 second
    lcd.clear();
    lcd.print("Ready for input");
  }
}
